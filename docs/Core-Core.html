lexer :: MonadThrow m => String -> m Token Source #

文字列を最小単位に分解します

>>> lexer "a b c"
L [A "a", A "b", A "c"]

parser :: (MonadThrow m, MonadState NameSpace m) => Token -> m IndexedTerm Source #

最小単位に分解された文字を

  1. 木に変換する
  2. 名前から解放して、de brujin indexingする
  3. 名前空間を生成する
>>> parser $ L [A "if", A "true", A "zero", A "false"]
(if true zero false)
